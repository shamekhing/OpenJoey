//	任意の要素をdeleteするための基底オブジェクト
class deleter_base {
public:
	virtual void	do_delete()=0;
	//	配列ポインタをdeleteする
};

//	任意の配列をdeleteするためのテンプレート
template<class T>
class array_deleter : public deleter_base {
public:
	array_deleter(T* pa) : m_pa(pa) {}
	virtual void	do_delete() {
		delete [] m_pa;
	}
private:
	T* m_pa;	//	削除すべきアドレスを保持しておく
};
