//	『蒼き大地』のメッセージボックスは、全角18文字×3行。
//	（禁則文字があればその行は19文字になる）

//	メッセージボックス用素材の描画座標および素材ファイル名
//	（複数指定できる）
#mes_parts	7 319 interface/talk1/window.yga

//	禁則文字を含まないメッセージボックスの最大幅
#mes_width	430
//	↑（『蒼き大地』では、有効幅470ドットで、
//	文字が24×24ドットなので、430〜445 + α）

//	メッセージテキストの表示位置
#mes_xy	149 361

//	メッセージボックスのテキストフォントのサイズ
#mes_fontsize 22
//	↑（『蒼き大地』では、文字が22

//	文字と文字の間隔
#mes_space 2
//	↑（『蒼き大地』では、2ドット

//	メッセージボックスの行間
#mes_Height 0

//	禁則文字列
#ProhibitString 「」『』《》。、，ー…“”！（）
//	↑ここに自由に禁則文字列を追加できる

//	ネームプレート表示座標
#NamePlate 15 325

//	フェースマーク表示座標
//#FaceMark  15 353
//	↑これを指定しなければフェイスマークは表示はされない

//	それぞれの定義ファイルを指定できる
#BG_DEFINE		cfg/bg_define.txt
#SE_DEFINE		cfg/se_define.txt
#BGM_DEFINE		cfg/bgm_define.txt
#SC_DEFINE		cfg/sc_define.txt
#FACE_DEFINE	cfg/face_define.txt
#NAME_DEFINE	cfg/name_define.txt
#BUTTON_DEFINE	interface/talk1/button_define.txt

#mes_buttonX	127 329 5
